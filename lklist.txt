//Function to check if the linked list has a loop.
    bool detectLoop(Node* head)
    {
        Node* low=head;
        Node* high=head;
        while(high!=NULL && high->next!=NULL){
            low=low->next;
            high=high->next->next;
            if(high==low){
                return true;
            }
        }
        return false;
    }
